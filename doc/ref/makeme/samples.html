MakeMe Samples

Here are some example MakeMe files in the Appweb and Ejscript products:

  • Appweb main.me — product level MakeMe file
  • Appweb src/server/server.me — MakeMe file to build appweb binary
  • Appweb src/libappweb.me — MakeMe file to build appweb library
  • Appweb src/modules/modules.me — MakeMe file to build appweb modules
  • Appweb test/test.me — MakeMe file to unit test appweb
  • Ejscript src/core/core.me — MakeMe file to build the ejscript module
  • Ejscript src/cmd/cmd.me — MakeMe file to build the ejscript executables

Example

Here is a sample main.me file:

/*
    main.me -- Main MakeMe File
 */
MakeMe.load({
    blend: [
        /* Blend these MakeMe files. Loaded before this file is processed */
        'company.me',
    ],
    customize: [
        /* Optionally load this if present to overwrite any property */
        'custom.me',
    ],
    '+modules': [
        /* Can put scripts or compiled mod file shere */
        'makeme/company.es',
    ],
    /*
        Core settings
     */
    settings: {
        product: 'rocket',
        title: 'Acme Rocket',
        company: 'Acme',
        version: '1.0.0',
        /* Required MakeMe version */
        me: '0.8.0'
        /* Do dynamic linking by default */
        static: false,
    },
    targets: {
        librocket: {
            /* Target to build a library */
            type: 'lib',
            sources: [
                /* Use all C source in rocket */
                'rocket/*.c'
                /* Use all C source anywhere under space */
                'space/**.c'
            ],
            /* Exclude any files with gemini or mercury in the name */
            exclude: /mercury|gemini/,
            /* Export these headers to the 'platform/inc' directory */
            headers:     [ 'rocket/rocket.h' ],
            /* Use these defines */
            '+defines':  [ 'ION_DRIVE', ],
            /* Search this directory for includes */
            '+includes': [ 'rocket', ],
            /* Add these compiler switches */
            '+compiler': [ '-Wall', ],
        },
        saturn: {
            /* Target to build an executable */
            type: 'exe',
            sources: [ 'saturn.c' ],
            depends: [ 'librocket'],
        },
        'saturn-gui': {
            /* Target to build an executable only on windows */
            enable: me.platform.os == 'windows',
            sources: [ 'saturnGui.c' ],
            depends: [ 'librocket'],
        },
    },
})
